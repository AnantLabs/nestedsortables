Left to Right nesting showing serialization.

//Code:
	$('#left-to-right').NestedSortable(
		{
			accept: 'page-item1',
			opacity: .8,
			helperclass: 'helper',
			onChange: function(serialized) {
				$('#left-to-right-ser')
				.html("This can be passed as parameter to a GET request: 
" 
						+ serialized[0].hash);
			}
		}
	);
